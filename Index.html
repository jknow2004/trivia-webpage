        <div id="question">Loading question...</div>

        <button onclick="fetchTrivia()">New Question</button>

    </div>


    <script>

        async function fetchTrivia() {

            try {

                const response = await fetch('https://opentdb.com/api.php?amount=1&type=multiple');

                const data = await response.json();

                const trivia = data.results[0];


                const question = decodeHtml(trivia.question);

                const answers = [...trivia.incorrect_answers, trivia.correct_answer].sort(() => Math.random() - 0.5);

                

                let answerList = answers.map(answer => `<li>${decodeHtml(answer)}</li>`).join('');

                

                document.getElementById('question').innerHTML = `

                    <p><strong>${question}</strong></p>

                    <ul>${answerList}</ul>

                `;

            } catch (error) {

                document.getElementById('question').innerHTML = 'Failed to load question. Try again!';

            }

        }


        function decodeHtml(html) {

            let txt = document.createElement('textarea');

            txt.innerHTML = html;

            return txt.value;

        }


        fetchTrivia();

    </script>
